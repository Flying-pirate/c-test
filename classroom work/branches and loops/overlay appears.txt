#define _CRT_SECURE_NO_WARNINGS
#include <stdio.h>
#include <string.h>

int main() {
	char arr1[] = {"Welcom to the Flying-pirate's world!!!!"};
	char arr2[] = {"#######################################" };
	int right = strlen(arr1)-1;
	int left = 0;
	while (left <= right) {
		arr2[left] = arr1[left];
		arr2[right] = arr1[right];
		left++;
		right--;
		printf("%s\n", arr2);//用“%c\n”出现错误
	}
	
	return 0;
}
